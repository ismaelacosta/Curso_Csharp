CREATE TABLE pets(
  id INT PRIMARY KEY AUTO_INCREMENT,
  name VARCHAR(50),
  owner VARCHAR(50), 
  species VARCHAR(50),
  sex CHAR(1),
  birth DATE,
  death DATE
);
